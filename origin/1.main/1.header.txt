<script>
document.addEventListener('DOMContentLoaded', function() {
    const section = document.querySelector('#s20251014418a368b55121');
    if (!section) {
        return;
    }

    const navItems = section.querySelectorAll('.nav-item[data-route]');
    const mobileBlocks = section.querySelectorAll('.mobile-nav-block[data-route]');

    function getCurrentRoute() {
        const path = window.location.pathname.toLowerCase();
        const url = window.location.href.toLowerCase();
        const hostname = window.location.hostname.toLowerCase();

        if (path.includes('/company') || url.includes('/company')) return 'company';
        if (path.includes('/process') || url.includes('/process')) return 'process';
        if (path.includes('/fund') || url.includes('/fund')) return 'fund';
        if (path.includes('/pro') || url.includes('/pro')) return 'pro';
        if (path.includes('/mkt') || url.includes('/mkt')) return 'mkt';
        
        // 홈은 특정 조건에서만 활성화
        if (path.includes('/home') ||
            (hostname === 'mjgood.kr' && (path === '/' || path === '')) ||
            (hostname === 'mjgood.imweb.me' && (path === '/' || path === ''))) {
            return 'home';
        }
        
        return null;
    }

    function updateActiveLinks() {
        const currentRoute = getCurrentRoute();

        navItems.forEach(item => {
            item.classList.toggle('active', item.dataset.route === currentRoute);
        });

        mobileBlocks.forEach(block => {
            block.classList.toggle('active', block.dataset.route === currentRoute);
        });
    }

    updateActiveLinks();
});
</script>
