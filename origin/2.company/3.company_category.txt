// 네비게이션 블록 클릭으로 카드 이동
function scrollToBusinessCardNew(index) {
    const businessGrid = document.querySelector('#s20250731337f4d2ea9683 .business-grid');
    const cards = document.querySelectorAll('#s20250731337f4d2ea9683 .business-card');
    const navBlocks = document.querySelectorAll('#s20250731337f4d2ea9683 .nav-block');
    const indicators = document.querySelectorAll('#s20250731337f4d2ea9683 .indicator-dot');
    
    if (businessGrid && cards[index]) {
        // 카드를 중앙에 정렬
        const card = cards[index];
        const cardWidth = card.offsetWidth;
        const containerWidth = businessGrid.offsetWidth;
        const cardLeft = card.offsetLeft;
        const scrollPosition = cardLeft - (containerWidth - cardWidth) / 2;
        
        businessGrid.scrollTo({
            left: scrollPosition,
            behavior: 'smooth'
        });
        
        // 네비 블록 활성화
        navBlocks.forEach((block, i) => {
            block.classList.toggle('active', i === index);
        });
        
        // 인디케이터도 업데이트
        indicators.forEach((dot, i) => {
            dot.classList.toggle('active', i === index);
        });
    }
}

// 모바일 스크롤 인디케이터 업데이트 및 클릭 기능
document.addEventListener('DOMContentLoaded', function() {
    if (window.innerWidth <= 768) {
        const businessGrid = document.querySelector('#s20250731337f4d2ea9683 .business-grid');
        const indicators = document.querySelectorAll('#s20250731337f4d2ea9683 .indicator-dot');
        const cards = document.querySelectorAll('#s20250731337f4d2ea9683 .business-card');
        const navBlocks = document.querySelectorAll('#s20250731337f4d2ea9683 .nav-block');
        
        if (businessGrid && indicators.length > 0 && cards.length > 0) {
            // 스크롤 이벤트 - 인디케이터와 네비 블록 업데이트
            businessGrid.addEventListener('scroll', function() {
                // 현재 중앙에 있는 카드 찾기
                const containerWidth = this.offsetWidth;
                const scrollLeft = this.scrollLeft;
                const center = scrollLeft + containerWidth / 2;
                
                let activeIndex = 0;
                cards.forEach((card, index) => {
                    const cardLeft = card.offsetLeft;
                    const cardRight = cardLeft + card.offsetWidth;
                    const cardCenter = cardLeft + card.offsetWidth / 2;
                    
                    // 카드의 중심이 컨테이너 중앙에 가장 가까운 카드 찾기
                    if (Math.abs(cardCenter - center) < Math.abs(cards[activeIndex].offsetLeft + cards[activeIndex].offsetWidth / 2 - center)) {
                        activeIndex = index;
                    }
                });
                
                // 인디케이터 업데이트
                indicators.forEach((dot, index) => {
                    dot.classList.toggle('active', index === activeIndex);
                });
                
                // 네비 블록 업데이트
                navBlocks.forEach((block, index) => {
                    block.classList.toggle('active', index === activeIndex);
                });
                
                // 카드 활성 상태 업데이트
                cards.forEach((card, i) => {
                    card.classList.toggle('active', i === activeIndex);
                });
            });
            
            // 인디케이터 클릭 이벤트
            indicators.forEach((dot, index) => {
                dot.addEventListener('click', function() {
                    scrollToBusinessCardNew(index);
                });
            });
        }
    }
});
